```python
"""Human-readable citation extraction and verification for agent memos."""
from __future__ import annotations

import re
from sqlalchemy import text
from sqlalchemy.ext.asyncio import AsyncSession

from app.core.legal.extractor import (
    SCC_PATTERN, SCC_ONLINE_PATTERN, AIR_PATTERN,
    INSC_PATTERN, SCR_PATTERN, CRLJ_PATTERN, SCALE_PATTERN,
)

_ALL_PATTERNS = [
    SCC_ONLINE_PATTERN,  # must be before SCC_PATTERN (more specific)
    SCC_PATTERN,
    AIR_PATTERN,
    INSC_PATTERN,
    SCR_PATTERN,
    CRLJ_PATTERN,
    SCALE_PATTERN,
]


def extract_citations_from_text(text_content: str) -> list[str]:
    """Extract all Indian legal citation strings from text."""
    citations: list[str] = []
    for pattern in _ALL_PATTERNS:
        for match in pattern.finditer(text_content):
            citations.append(match.group(0))
    return list(set(citations))


async def verify_citations_against_db(
    citations: list[str],
    db: AsyncSession,
) -> tuple[list[str], list[str]]:
    """Check which citations exist in the database.

    Returns (verified, unverified) citation lists.
    """
    if not citations:
        return [], []

    verified: list[str] = []
    unverified: list[str] = []

    for cite in citations:
        # Check both cases.citation and case_citation_equivalents
        result = await db.execute(
            text(
                "SELECT COUNT(*) FROM cases WHERE citation ILIKE :cite "
                "UNION ALL "
                "SELECT COUNT(*) FROM case_citation_equivalents WHERE citation_text ILIKE :cite"
            ),
            {"cite": f"%{cite}%"},
        )
        counts = [row[0] for row in result.fetchall()]
        if any(c > 0 for c in counts):
            verified.append(cite)
        else:
            unverified.append(cite)

    return verified, unverified


def check_grounding(
    memo_citations: list[str],
    search_result_citations: list[str],
) -> list[str]:
    """Find citations in memo that were NOT in the search results (potentially hallucinated)."""
    search_set = {c.strip().lower() for c in search_result_citations}
    return [c for c in memo_citations if c.strip().lower() not in search_set]
```

async def _reformulate_query(
    question: str,
    history: list[dict],
    llm: LLMProvider,
) -> str:
    """Reformulate a follow-up question using conversation context."""
    if not history:
        return question

    # Build concise history summary (last 4 messages)
    recent = history[-4:]
    history_text = "\n".join(
        f"{'User' if m['role'] == 'user' else 'Assistant'}: {m['content'][:200]}"
        for m in recent
    )

    prompt = (
        f"Given this conversation:\n{history_text}\n\n"
        f"The user now asks: \"{question}\"\n\n"
        f"Rewrite the user's question as a standalone search query that includes "
        f"the necessary context from the conversation. Return ONLY the reformulated query, "
        f"nothing else. Keep it concise (under 50 words)."
    )

    try:
        reformulated = await llm.generate(prompt)
        return reformulated.strip() if reformulated.strip() else question
    except Exception:
        return question

"""Detect citation treatment language in judgment text."""
from __future__ import annotations

import re

_TREATMENT_PATTERNS = {
    "OVERRULED": [
        re.compile(r"\b(overruled|overrule|no longer good law|per incuriam)\b", re.IGNORECASE),
    ],
    "DISTINGUISHED": [
        re.compile(r"\b(distinguishable|distinguished|distinguish)\b", re.IGNORECASE),
    ],
    "AFFIRMED": [
        re.compile(r"\b(affirmed|affirm|approved|followed|reaffirmed)\b", re.IGNORECASE),
    ],
    "EXPLAINED": [
        re.compile(r"\b(explained|clarified|interpreted)\b", re.IGNORECASE),
    ],
}


def detect_citation_treatment(text: str) -> list[dict]:
    """Detect treatment language in judgment text.

    Returns list of {"treatment": str, "context": str} dicts.
    """
    treatments: list[dict] = []
    for treatment_type, patterns in _TREATMENT_PATTERNS.items():
        for pattern in patterns:
            for match in pattern.finditer(text):
                # Extract surrounding context (100 chars each side)
                start = max(0, match.start() - 100)
                end = min(len(text), match.end() + 100)
                context = text[start:end].strip()
                treatments.append({
                    "treatment": treatment_type,
                    "keyword": match.group(0),
                    "context": context,
                })
    return treatments
```

function stripSourcesSection(content: string): string {
    // Remove the LLM-generated "Sources:" section (we show our own source badges)
    const sourcesIdx = content.lastIndexOf("\nSources:");
    if (sourcesIdx > 0) return content.slice(0, sourcesIdx).trim();
    return content;
}

function processInlineCitations(
    children: React.ReactNode,
    sources: ChatSource[],
): React.ReactNode {
    if (!children || !sources?.length) return children;
    if (typeof children !== "string") return children;

    // Replace [1], [2] etc with clickable links
    return children.split(/(\[\d+\])/).map((part, i) => {
        const match = part.match(/^\[(\d+)\]$/);
        if (match) {
            const idx = parseInt(match[1]) - 1;
            const source = sources[idx];
            if (source) {
                return (
                    <a key={i} href={`/case/${source.case_id}`}
                       className="text-[var(--gold)] font-semibold hover:underline cursor-pointer"
                       title={source.citation || source.title || "View case"}>
                        {part}
                    </a>
                );
            }
        }
        return part;
    });
}
```
