```python
def normalize_issue_tags(tags: list[str] | None) -> list[str]:
    """Normalize issue classification tags to canonical taxonomy.

    - Maps known variant tags to their canonical form
    - Preserves unknown tags as-is (no data loss)
    - Deduplicates after normalization
    """
    if not tags:
        return []
    seen: set[str] = set()
    result: list[str] = []
    for tag in tags:
        canonical = NORMALIZATION_MAP.get(tag, tag)
        if canonical not in seen:
            seen.add(canonical)
            result.append(canonical)
    return result


def get_category_for_tag(tag: str) -> str | None:
    """Return the display category name for a tag, or None if unknown."""
    if tag in _TAG_TO_CATEGORY:
        return _TAG_TO_CATEGORY[tag]
    # Try matching by prefix (e.g., "criminal_law.some_new_thing" → "Criminal Law")
    prefix = tag.split(".")[0]
    return _CATEGORY_KEY_TO_NAME.get(prefix)


def get_all_subtopics(category: str) -> dict[str, str]:
    """Return {tag: label} for all subtopics in a category."""
    return LEGAL_TAXONOMY.get(category, {})


def get_categories() -> list[str]:
    """Return all category display names."""
    return list(LEGAL_TAXONOMY.keys())


def get_taxonomy_prompt_text() -> str:
    """Generate the taxonomy list for inclusion in LLM prompts."""
    lines = ["Classify using ONLY tags from this taxonomy:\n"]
    for category, subtopics in LEGAL_TAXONOMY.items():
        lines.append(f"\n{category}:")
        for tag, label in subtopics.items():
            lines.append(f"  - {tag} ({label})")
    return "\n".join(lines)
```

```python
async def fetch_pg_metadata(db_session) -> list[dict]:
    """Fetch enrichment fields from PostgreSQL for all cases."""
    from sqlalchemy import text

    result = await db_session.execute(text("""
        SELECT id, jurisdiction, coram_size, is_reportable, opinion_type,
               issue_classification, primary_legal_issue, fact_pattern_summary,
               headnotes
        FROM cases
        WHERE id IS NOT NULL
    """))
    return [dict(r._mapping) for r in result.fetchall()]


async def fetch_statute_interpretations(db_session) -> list[dict]:
    """Fetch statute section interpretations from PostgreSQL."""
    from sqlalchemy import text

    result = await db_session.execute(text("""
        SELECT c.id AS case_id, si.value->>'section' AS section,
               si.value->>'act' AS act,
               si.value->>'interpretation_summary' AS interpretation_summary
        FROM cases c,
             jsonb_array_elements(c.statute_sections_interpreted) AS si(value)
        WHERE c.statute_sections_interpreted IS NOT NULL
          AND c.statute_sections_interpreted::text != '[]'
    """))
    return [dict(r._mapping) for r in result.fetchall()]


async def enrich_neo4j_from_postgres(
    graph_store,
    pg_rows: list[dict],
) -> int:
    """Write PostgreSQL metadata to Neo4j Case node properties."""
    batch: list[dict] = []
    for row in pg_rows:
        # Normalize issue tags
        raw_tags = row.get("issue_classification") or []
        normalized = normalize_issue_tags(raw_tags)

        # Extract first headnote text
        headnote_text = None
        raw_headnotes = row.get("headnotes")
        if raw_headnotes:
            try:
                hn = _json.loads(raw_headnotes) if isinstance(raw_headnotes, str) else raw_headnotes
                if hn and isinstance(hn, list) and len(hn) > 0:
                    headnote_text = (hn[0].get("proposition") or "")[:500]
            except (ValueError, TypeError):
                pass

        batch.append({
            "id": row["id"],
            "jurisdiction": row.get("jurisdiction") or "",
            "coram_size": row.get("coram_size") or 0,
            "is_reportable": bool(row.get("is_reportable")),
            "opinion_type": row.get("opinion_type") or "",
            "issue_tags": ", ".join(normalized),
            "primary_legal_issue": (row.get("primary_legal_issue") or "")[:200],
            "fact_pattern_summary": (row.get("fact_pattern_summary") or "")[:500],
            "headnote_text": headnote_text or "",
        })

    # Write in batches of 500
    written = 0
    for i in range(0, len(batch), 500):
        chunk = batch[i:i + 500]
        await graph_store.query(
            cypher=(
                "UNWIND $rows AS row "
                "MATCH (c:Case {id: row.id}) "
                "SET c.jurisdiction = row.jurisdiction, "
                "    c.coram_size = row.coram_size, "
                "    c.is_reportable = row.is_reportable, "
                "    c.opinion_type = row.opinion_type, "
                "    c.issue_tags = row.issue_tags, "
                "    c.primary_legal_issue = row.primary_legal_issue, "
                "    c.fact_pattern_summary = row.fact_pattern_summary, "
                "    c.headnote_text = row.headnote_text"
            ),
            params={"rows": chunk},
        )
        written += len(chunk)
    return written


async def create_issue_topic_nodes(
    graph_store,
    pg_rows: list[dict],
) -> int:
    """Create IssueTopic nodes and CLASSIFIED_AS edges."""
    # Collect unique tags and case-tag pairs
    edges: list[dict] = []
    for row in pg_rows:
        raw_tags = row.get("issue_classification") or []
        normalized = normalize_issue_tags(raw_tags)
        for tag in normalized:
            category = get_category_for_tag(tag) or "Other"
            subtopic = tag.split(".", 1)[1] if "." in tag else tag
            edges.append({
                "case_id": row["id"],
                "tag": tag,
                "category": category,
                "subtopic": subtopic,
            })

    if not edges:
        return 0

    # Batch MERGE IssueTopic nodes + CLASSIFIED_AS edges
    written = 0
    for i in range(0, len(edges), 500):
        chunk = edges[i:i + 500]
        await graph_store.query(
            cypher=(
                "UNWIND $rows AS row "
                "MERGE (t:IssueTopic {id: row.tag}) "
                "ON CREATE SET t.category = row.category, t.subtopic = row.subtopic "
                "WITH t, row "
                "MATCH (c:Case {id: row.case_id}) "
                "MERGE (c)-[:CLASSIFIED_AS]->(t)"
            ),
            params={"rows": chunk},
        )
        written += len(chunk)
    return written


async def create_statute_section_nodes(
    graph_store,
    statute_rows: list[dict],
) -> int:
    """Create StatuteSection nodes and INTERPRETS edges."""
    if not statute_rows:
        return 0

    # Normalize section IDs
    edges: list[dict] = []
    for row in statute_rows:
        section = row.get("section") or ""
        act = row.get("act") or ""
        section_id = f"{act}_{section}".lower().replace(" ", "_").replace(",", "")[:100]
        edges.append({
            "case_id": row["case_id"],
            "section_id": section_id,
            "section": section,
            "act": act,
            "interpretation": (row.get("interpretation_summary") or "")[:500],
        })

    written = 0
    for i in range(0, len(edges), 500):
        chunk = edges[i:i + 500]
        await graph_store.query(
            cypher=(
                "UNWIND $rows AS row "
                "MERGE (s:StatuteSection {id: row.section_id}) "
                "ON CREATE SET s.section = row.section, s.act = row.act "
                "WITH s, row "
                "MATCH (c:Case {id: row.case_id}) "
                "MERGE (c)-[r:INTERPRETS]->(s) "
                "SET r.interpretation = row.interpretation"
            ),
            params={"rows": chunk},
        )
        written += len(chunk)
    return written
```

```python
async def get_subtopics(
    *,
    graph_store: GraphStore,
    category: str | None = None,
    redis_client=None,
) -> list[dict]:
    """Get IssueTopic nodes with case counts, optionally filtered by category."""
    cache_key = f"graph:subtopics:{category or 'all'}"
    if redis_client is not None:
        try:
            cached = await redis_client.get(cache_key)
            if cached is not None:
                return json.loads(cached)
        except Exception:
            pass

    category_filter = "WHERE t.category = $category " if category else ""
    params = {"category": category} if category else {}

    records = await graph_store.query(
        cypher=(
            f"MATCH (c:Case)-[:CLASSIFIED_AS]->(t:IssueTopic) {category_filter}"
            "RETURN t.id AS tag, t.category AS category, t.subtopic AS subtopic, "
            "  count(c) AS count "
            "ORDER BY count DESC "
            "LIMIT 50"
        ),
        params=params,
    )

    result = [
        {"tag": r["tag"], "category": r["category"], "subtopic": r["subtopic"], "count": r["count"]}
        for r in records
    ]

    if redis_client is not None:
        try:
            await redis_client.setex(cache_key, 3600, json.dumps(result))
        except Exception:
            pass

    return result


async def get_statute_sections(
    *,
    graph_store: GraphStore,
    redis_client=None,
) -> list[dict]:
    """Get StatuteSection nodes with case counts."""
    cache_key = "graph:statute_sections"
    if redis_client is not None:
        try:
            cached = await redis_client.get(cache_key)
            if cached is not None:
                return json.loads(cached)
        except Exception:
            pass

    records = await graph_store.query(
        cypher=(
            "MATCH (c:Case)-[:INTERPRETS]->(s:StatuteSection) "
            "RETURN s.id AS id, s.act AS act, s.section AS section, "
            "  count(c) AS count "
            "ORDER BY count DESC "
            "LIMIT 50"
        ),
    )

    result = [
        {"id": r["id"], "act": r["act"], "section": r["section"], "count": r["count"]}
        for r in records
    ]

    if redis_client is not None:
        try:

            pass

    return result
```
